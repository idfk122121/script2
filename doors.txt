local function ExecuteLoadStringFromURL(scriptUrl)
    local loadedScript = loadstring(game:HttpGet(scriptUrl))
    if loadedScript then
        local success, error = pcall(loadedScript)
        if not success then
            warn("Error executing script from URL:", error)
        end
    else
        warn("Failed to load script from URL:", scriptUrl)
    end
end
